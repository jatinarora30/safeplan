1 """
2 @file visualize.py
3 @brief Render 2D/3D occupancy grids and algorithm paths from run outputs.
4 
5 @details
6 Provides convenient plotting for benchmark runs. For 2D grids, writes a PNG
7 with obstacles, start/goal markers, and optional path overlays. For 3D grids,
8 writes an interactive Plotly HTML (isosurface + point cloud + paths), and can
9 optionally save a Matplotlib 3D PNG snapshot when available.
10 
11 Expected per-iteration input (for each algorithm):
12 - Located at `outputs/<runDetails>/<algo>/iter_<N>.json`
13 - Contains:
14  - `envInfo.envName` : string name of the environment (e.g., "house_2d", "maze3d")
15  - `startGoal.start` : start index tuple/list
16  - `startGoal.goal` : goal index tuple/list
17  - `pathInfo.path` : list of coordinates (tuples/lists)
18 
19 Environment grid source (single, shared across algos for the same iter):
20 - Located at `configs/generate_grid/{envName}/{envName}_grid.json`
21 - Contains:
22  - `grid` : N-D list/array (0=free, >0=occupied)
23 
24 Artifacts:
25 - 2D PNG: `results/viz/{envName}_grid2d_iter<N>.png`
26 - 3D HTML: `results/viz/{envName}_grid3d_iter<N>.html`
27 - Optional 3D PNG (fallback snapshot): `results/viz/{envName}_grid3d_iter<N>.png`
28 
29 @note
30 - Plotly is preferred for 3D (opens in a browser). If Plotly is unavailable,
31  the code attempts a Matplotlib 3D fallback when possible.
32 - Obstacles are rendered as black (2D) or red (3D). Paths are colored and
33  labeled by algorithm name.
34 - The function auto-creates `results/viz/` if it does not exist.
35 - Start/Goal are ALWAYS taken from the first algorithm's iter file.
36 - The GRID is ALWAYS taken from `configs/generate_grid/{envName}/{envName}_grid.json`, where
37  `envName` is read from the first algorithm's iter file (`envInfo.envName`) and lowercased.
38 
39 @see plotly.graph_objects, matplotlib.pyplot
40 """

82 def _plot_3d_plotly_combo(occ_bool, paths, start, goal, title, html_path,
83  show_paths=True, max_points=120_000):
84  """
85  @brief Render a 3D scene with Plotly and write an HTML file.
86 
87  @details
88  Draws occupied voxels using an isosurface plus a downsampled point cloud to
89  guarantee visibility. Overlays optional algorithm paths and start/goal markers.
90  Always writes an HTML file and prints its path.
91 
92  @param occ_bool numpy.ndarray
93  Boolean occupancy mask (True = obstacle).
94  @param paths dict[str, list[tuple]]
95  Mapping from algorithm label to path (sequence of 3D points).
96  @param start tuple[int, int, int]
97  Start index in the 3D grid.
98  @param goal tuple[int, int, int]
99  Goal index in the 3D grid.
100  @param title str
101  Figure title.
102  @param html_path str
103  Output HTML file path.
104  @param show_paths bool
105  Whether to overlay paths on the plot.
106  @param max_points int
107  Maximum number of obstacle points to plot in the point-cloud overlay.
108  """
109  occ_u8 = occ_bool.astype(np.uint8)
110  fig = go.Figure()
111 
112  # Isosurface of occupied voxels
113  fig.add_trace(go.Isosurface(
114  value=occ_u8,
115  isomin=0.5, isomax=1.0, surface_count=1,
116  opacity=0.55,
117  colorscale=[[0, "red"], [1, "red"]],
118  caps=dict(x_show=False, y_show=False, z_show=False),
119  showscale=False,
120  lighting=dict(ambient=0.5, diffuse=0.7, specular=0.2, roughness=0.8, fresnel=0.1),
121  lightposition=dict(x=1000, y=1000, z=1000),
122  ))
123 
124  # Downsampled point cloud overlay (guaranteed visible)
125  coords = np.argwhere(occ_bool) # (N,3)
126  if len(coords) > 0:
127  step = max(1, len(coords) // max_points)
128  coords = coords[::step]
129  fig.add_trace(go.Scatter3d(
130  x=coords[:, 0], y=coords[:, 1], z=coords[:, 2],
131  mode="markers", name="Obstacles",
132  marker=dict(size=2, color="red", opacity=0.8)
133  ))
134 
135  # Paths
136  if show_paths and paths:
137  for label, path in paths.items():
138  if not path:
139  continue
140  a = np.asarray(path)
141  fig.add_trace(go.Scatter3d(
142  x=a[:, 0], y=a[:, 1], z=a[:, 2],
143  mode="lines", name=label, line=dict(width=6)
144  ))
145 
146  # Start / Goal
147  fig.add_trace(go.Scatter3d(
148  x=[start[0]], y=[start[1]], z=[start[2]],
149  mode="markers", name="Start",
150  marker=dict(size=7, color="yellow", line=dict(width=2, color="black"))
151  ))
152  fig.add_trace(go.Scatter3d(
153  x=[goal[0]], y=[goal[1]], z=[goal[2]],
154  mode="markers", name="Goal",
155  marker=dict(size=7, color="red", line=dict(width=2, color="black"))
156  ))
157 
158  fig.update_layout(
159  title=title,
160  scene=dict(xaxis_title="X", yaxis_title="Y", zaxis_title="Z", aspectmode="data"),
161  margin=dict(l=0, r=0, b=0, t=40),
162  legend=dict(itemsizing="constant")
163  )
164 
165  fig.write_html(html_path, auto_open=True)
166  print(f"[viz] Wrote 3D HTML to: {html_path}")
