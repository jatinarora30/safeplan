115  def makeConvexHull(self, vertices) -> tuple:
116  """
117  @brief Convert a set of polygon vertices to half-space form.
118  @param vertices Iterable of vertex coordinates (shape: N x D).
119  @return (A, b) where A @ x <= b describes the convex hull.
120  @details
121  Uses @c scipy.spatial.ConvexHull to compute plane equations in the
122  form [A | c] so that A x + c = 0 for faces. We rearrange to A x <= b
123  with b = -c.
124  @warning Vertices must define a valid (non-degenerate) convex polygon/polytope.
125  """
126  hull = ConvexHull(vertices)
127  A = hull.equations[:, :-1]
128  b = -hull.equations[:, -1]
129  return (A, b)
